// 示例算法：动态规划入口
int solve(int n) {
if (n <= 1) return 1;
vector<int> dp(n + 1, 0);
dp[0] = 1; dp[1] = 1;
for (int i = 2; i <= n; ++i) {
dp[i] = dp[i - 1] + dp[i - 2];
}
return dp[n];
}
