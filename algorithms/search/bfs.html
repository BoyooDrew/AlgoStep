// 广度优先搜索 - Breadth First Search
// 使用邻接矩阵 + 队列实现
int adj[10][10];
int visited[10];
int queue[100];
int head = 0, tail = 0;
void bfs(int start, int n) {
visited[start] = 1;
queue[tail++] = start;
while (head < tail) {
int cur = queue[head++];
for (int i = 0; i < n; i++) {
if (adj[cur][i] == 1 && !visited[i]) {
visited[i] = 1;
queue[tail++] = i;
}
}
}
}
