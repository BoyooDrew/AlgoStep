// 广度优先搜索 - Breadth First Search
// 时间复杂度: O(V+E) 空间复杂度: O(V)
// 使用邻接矩阵 + 队列实现
#include <bits/stdc++.h>
using namespace std;
int adj[10][10]; // 邻接矩阵
int visited[10]; // 访问标记数组
int queue[100]; // 模拟队列
int head = 0, tail = 0;
void bfs(int start, int n) {
visited[start] = 1; // 标记起始节点
queue[tail++] = start; // 起始节点入队
while (head < tail) { // 队列非空
int cur = queue[head++]; // 出队当前节点
for (int i = 0; i < n; i++) { // 遍历邻居
if (adj[cur][i] == 1 && !visited[i]) {
visited[i] = 1; // 标记邻居已访问
queue[tail++] = i; // 邻居入队
}
}
}
cout << "BFS 完成" << endl;
}
int main() {
int n = 6;
memset(visited, 0, sizeof(visited));
bfs(0, n); // 从节点 0 开始搜索
return 0;
}
