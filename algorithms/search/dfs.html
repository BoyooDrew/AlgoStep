// 深度优先搜索 - Depth First Search
// 时间复杂度: O(V+E) 空间复杂度: O(V)
// 使用邻接矩阵存储无向图
#include <bits/stdc++.h>
using namespace std;
int adj[10][10]; // 邻接矩阵
int visited[10]; // 访问标记数组
void dfs(int node, int n) {
visited[node] = 1; // 标记当前节点已访问
for (int i = 0; i < n; i++) { // 遍历所有可能的邻居
if (adj[node][i] == 1 && !visited[i]) {
dfs(i, n); // 递归访问未访问的邻居
}
}
cout << "回溯: 离开节点 " << node << endl;
// 回溯: 从 node 返回到上一层
}
int main() {
int n = 6;
memset(visited, 0, sizeof(visited));
dfs(0, n); // 从节点 0 开始搜索
return 0;
}
