// 深度优先搜索 - Depth First Search
// 使用邻接矩阵存储图
int adj[10][10];
int visited[10];
void dfs(int node, int n) {
visited[node] = 1;
// 处理当前节点...
for (int i = 0; i < n; i++) {
if (adj[node][i] == 1 && !visited[i]) {
dfs(i, n);
}
}
// 回溯: 从 node 退出
}
