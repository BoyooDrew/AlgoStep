// 快速排序 - Quick Sort
// 平均时间复杂度: O(n log n) 空间复杂度: O(log n)
// 思想: 分治法 - 选基准值 (pivot), 将数组分成两部分, 递归排序
void quickSort(int arr[], int left, int right) {
if (left >= right) return; // 递归终止条件 (base case)
// 分区操作 partition
int pivot = arr[right]; // 基准值 pivot (选最右元素)
int i = left;
for (int j = left; j < right; j++) { // 扫描元素
if (arr[j] < pivot) { // 比基准值小 → 放左边
int temp = arr[i];
arr[i] = arr[j];
arr[j] = temp;
i++;
}
}
// pivot 归位
int temp = arr[i];
arr[i] = arr[right];
arr[right] = temp;
// 此时: arr[left..i-1] < pivot <= arr[i+1..right]
quickSort(arr, left, i - 1); // 递归排序左半部分 (recursion)
quickSort(arr, i + 1, right); // 递归排序右半部分 (recursion)
}
