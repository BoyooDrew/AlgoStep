// 冒泡排序 - Bubble Sort
// 时间复杂度: O(n^2) 空间复杂度: O(1)
// 思想: 相邻元素两两比较，较大的元素逐步"冒泡"到末尾
void bubbleSort(int arr[], int n) {
for (int i = 0; i < n - 1; i++) { // 外层循环: 轮数(趟数 pass)
for (int j = 0; j < n - 1 - i; j++) { // 内层循环: 两两比较
if (arr[j] > arr[j + 1]) { // 比较相邻元素
int temp = arr[j]; // 交换 swap
arr[j] = arr[j + 1];
arr[j + 1] = temp;
}
}
// 第 i 趟结束, arr[n-1-i] 已确定
}
}
