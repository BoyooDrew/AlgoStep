// 插入排序 - Insertion Sort
// 时间复杂度: O(n^2) 空间复杂度: O(1)
// 思想: 将未排序元素逐个插入到已排序部分的正确位置
void insertionSort(int arr[], int n) {
for (int i = 1; i < n; i++) { // 从第2个元素开始
int key = arr[i]; // 当前待插入元素 (key)
int j = i - 1;
while (j >= 0 && arr[j] > key) { // 找插入位置 (向前查找)
arr[j + 1] = arr[j]; // 元素后移
j--;
}
arr[j + 1] = key; // 插入到正确位置
// arr[0..i] 已排序
}
}
