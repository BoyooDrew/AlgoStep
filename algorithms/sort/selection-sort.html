// 选择排序 - Selection Sort
// 时间复杂度: O(n^2) 空间复杂度: O(1)
// 思想: 每轮从未排序部分选出最小元素, 放到已排序部分末尾
void selectionSort(int arr[], int n) {
for (int i = 0; i < n - 1; i++) { // 外层循环: 轮数
int minIdx = i; // 假设当前位置最小
for (int j = i + 1; j < n; j++) { // 在未排序部分找最小值
if (arr[j] < arr[minIdx]) {
minIdx = j; // 更新最小值索引
}
}
if (minIdx != i) { // 交换最小值到正确位置
int temp = arr[i];
arr[i] = arr[minIdx];
arr[minIdx] = temp;
}
// 第 i 轮结束, arr[0..i] 已排序
}
}
