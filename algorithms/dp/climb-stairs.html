// 台阶问题动态规划算法
int climbStairs(int n) {
if (n <= 2) return n;
int dp[n+1];
dp[0] = 0; // 没有台阶
dp[1] = 1; // 只有1个台阶
dp[2] = 2; // 2个台阶
for (int i = 3; i <= n; i++) {
dp[i] = dp[i-1] + dp[i-2];
}
return dp[n];
}
