// 最长递增子序列 - Longest Increasing Subsequence
// dp[i]: 以 arr[i] 结尾的最长递增子序列长度
// 状态转移方程: dp[i] = max(dp[j] + 1), j < i 且 arr[j] < arr[i]
int lis(int arr[], int n) {
int dp[105];
int ans = 1;
for (int i = 0; i < n; i++) {
dp[i] = 1;
for (int j = 0; j < i; j++) {
if (arr[j] < arr[i]) {
if (dp[j] + 1 > dp[i])
dp[i] = dp[j] + 1;
}
}
if (dp[i] > ans) ans = dp[i];
}
return ans;
}
