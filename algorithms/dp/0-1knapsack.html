// 0-1背包问题 - 0-1 Knapsack Problem
// dp[i][j]: 前i个物品、背包容量为j时的最大价值
// 状态转移方程: dp[i][j] = max(dp[i-1][j], dp[i-1][j-w[i]] + v[i])
int dp[105][10005];
int w[105]; // 重量 weight
int v[105]; // 价值 value
int knapsack(int n, int W) {
for (int i = 0; i <= n; i++)
dp[i][0] = 0;
for (int j = 0; j <= W; j++)
dp[0][j] = 0;
for (int i = 1; i <= n; i++) {
for (int j = 1; j <= W; j++) {
if (j >= w[i]) {
dp[i][j] = dp[i-1][j];
if (dp[i-1][j-w[i]] + v[i] > dp[i][j])
dp[i][j] = dp[i-1][j-w[i]] + v[i];
} else {
dp[i][j] = dp[i-1][j];
}
}
}
return dp[n][W];
}
