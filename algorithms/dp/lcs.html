// 最长公共子序列 (LCS) 动态规划解法（基础数组版）
int longestCommonSubseq(string A, string B) {
int n = A.length(), m = B.length();
int dp[100][100] = {0};
for (int i = 1; i <= n; i++)
for (int j = 1; j <= m; j++)
if (A[i - 1] == B[j - 1])
dp[i][j] = dp[i - 1][j - 1] + 1;
else
dp[i][j] = max(dp[i - 1][j], dp[i][j - 1]);
return dp[n][m];
}
